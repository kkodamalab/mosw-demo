function showEmoji(emoji, x, y) { const el = document.createElement("div"); el.className = "emoji"; el.textContent = emoji; const w = window.innerWidth; const h = window.innerHeight; el.style.left = `${x * w}px`; el.style.top = `${y * h}px`; screen.appendChild(el); setTimeout(() => el.remove(), 3000); } onChildAdded(ref(db, "reactions"), snapshot => { const data = snapshot.val(); if (!data || !data.emoji) return; if (typeof data.x === "number" && typeof data.y === "number") { showEmoji(data.emoji, data.x, data.y); } else if (data.quadrant) { const fallback = { NW: [0.25, 0.25], NE: [0.75, 0.25], SW: [0.25, 0.75], SE: [0.75, 0.75] }; const [x, y] = fallback[data.quadrant] || [0.5, 0.5]; showEmoji(data.emoji, x, y); } });
